31 struct bootp {

36  u_int32_t bp_xid; /* transaction ID */
37  u_int16_t bp_secs; /* seconds since boot began */
38  u_int16_t bp_flags; /* flags: 0x8000 is broadcast */
39  struct in_addr bp_ciaddr; /* client IP address */

43  u_int8_t bp_chaddr[16]; /* client hardware address */
44  u_int8_t bp_sname[64]; /* server host name */

47 };
